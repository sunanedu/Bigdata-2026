## ตัวอย่าง SQL Query สำหรับการวิเคราะห์

```sql
-- 1. จำนวนอุบัติเหตุและผู้เสียชีวิตรายจังหวัด
SELECT จังหวัด, ภูมิภาค, 
       COUNT(*) AS จำนวนเหตุ,
       SUM(จำนวนผู้เสียชีวิต) AS ผู้เสียชีวิต,
       SUM(จำนวนผู้บาดเจ็บ) AS ผู้บาดเจ็บ,
       ROUND(AVG(ค่าใช้จ่ายรวม_บาท),0) AS ค่าใช้จ่ายเฉลี่ย
FROM road_accidents
GROUP BY จังหวัด, ภูมิภาค
ORDER BY จำนวนเหตุ DESC;

-- 2. อุบัติเหตุรายชั่วโมง จำแนกตามวันทำงาน/วันหยุด
SELECT ช่วงเวลา, วันหยุด_วันปกติ, COUNT(*) AS จำนวน
FROM road_accidents
GROUP BY ช่วงเวลา, วันหยุด_วันปกติ
ORDER BY ช่วงเวลา;

-- 3. ความสัมพันธ์ระหว่างแอลกอฮอล์และความรุนแรง
SELECT ระดับแอลกอฮอล์_g_dL, ความรุนแรง, COUNT(*) AS จำนวน,
       ROUND(COUNT(*)*100.0/SUM(COUNT(*)) OVER(PARTITION BY ระดับแอลกอฮอล์_g_dL),1) AS สัดส่วนpct
FROM road_accidents
GROUP BY ระดับแอลกอฮอล์_g_dL, ความรุนแรง;

-- 4. Top 10 สาเหตุที่ทำให้เสียชีวิตมากที่สุด
SELECT สาเหตุหลักของอุบัติเหตุ, 
       COUNT(*) AS จำนวนเหตุ,
       SUM(จำนวนผู้เสียชีวิต) AS ผู้เสียชีวิต
FROM road_accidents
WHERE จำนวนผู้เสียชีวิต > 0
GROUP BY สาเหตุหลักของอุบัติเหตุ
ORDER BY ผู้เสียชีวิต DESC
LIMIT 10;

-- 5. อัตราการเกิดเหตุต่อประชากร 100,000 คน
SELECT จังหวัด, COUNT(*) AS จำนวนเหตุ, ประชากรจังหวัด,
       ROUND(COUNT(*)*100000.0/ประชากรจังหวัด, 2) AS อัตราต่อแสนประชากร
FROM road_accidents
GROUP BY จังหวัด, ประชากรจังหวัด
ORDER BY อัตราต่อแสนประชากร DESC;
```
